How to find sum of two numbers in c++
#include
using namespace std;
int main()
{
    int a,b,sum;
    cout<<"Enter the 1st number\n";
    cin>>a;
    cout<<"Enter the 2nd number\n";
    cin>>b;
    sum=a+b;
    cout<<"Sum of two number "<<sum;
    return 0;
}
